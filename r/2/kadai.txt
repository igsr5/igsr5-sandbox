# 1
# 上から順に wの係数, xの係数, yの係数, zの係数
# 少し直感的じゃ無いので注意
ma<-matrix(
           c(
             2, -1, 1, -3,
             1, 2, -3, 1,
             3, -2, 1, 1,
             1, -3, 2, 4
           ),
           4,
           4
)

vb<-c(-2, 5, -7, 11)
solve(ma, vb)
